// ============================================================
//  Smart Parking Simulator
//  Using Arduino Nano + 7404 + 7408
// ============================================================

#include <Servo.h>

// ── IR Sensor Inputs ─────────────────────────────────────────
const int IR1 = A0;
const int IR2 = A1;
const int IR3 = A2;
const int IR4 = A3;

// ── Arduino → 7404 NOT Gate ─────────────────────────────────
const int S1_OUT = 2;
const int S2_OUT = 3;
const int S3_OUT = 4;
const int S4_OUT = 5;

// ── Ultrasonic Sensor ────────────────────────────────────────
const int TRIG = 8;
const int ECHO = 7;

// ── Car Signal → 7408 ───────────────────────────────────────
const int CAR_OUT = 6;

// ── FULL Signal ← 7408 ──────────────────────────────────────
const int FULL_IN = 11;

// ── Outputs ──────────────────────────────────────────────────
const int SERVO_PIN  = 9;
const int BUZZER_PIN = 10;

// ── Constants ────────────────────────────────────────────────
const int GATE_THRESHOLD = 20;

// Reversed Servo Direction
const int SERVO_CLOSED = 90;
const int SERVO_OPEN   = 0;

const int BARRIER_DELAY = 3000;

// ── Servo Object ─────────────────────────────────────────────
Servo barrier;

// ============================================================
// SETUP
// ============================================================

void setup() {

  Serial.begin(9600);

  // Outputs to 7404
  pinMode(S1_OUT, OUTPUT);
  pinMode(S2_OUT, OUTPUT);
  pinMode(S3_OUT, OUTPUT);
  pinMode(S4_OUT, OUTPUT);

  // Car signal to 7408
  pinMode(CAR_OUT, OUTPUT);

  // Ultrasonic
  pinMode(TRIG, OUTPUT);
  pinMode(ECHO, INPUT);

  // FULL signal from 7408
  pinMode(FULL_IN, INPUT);

  // Buzzer
  pinMode(BUZZER_PIN, OUTPUT);

  // Servo
  barrier.attach(SERVO_PIN);

  // Initial state
  barrier.write(SERVO_CLOSED);

  digitalWrite(BUZZER_PIN, LOW);

  delay(1000);

  Serial.println("Smart Parking System Started");
}

// ============================================================
// MAIN LOOP
// ============================================================

void loop() {

  // Read IR sensors
  int s1 = digitalRead(IR1);
  int s2 = digitalRead(IR2);
  int s3 = digitalRead(IR3);
  int s4 = digitalRead(IR4);

  // Send raw signals to 7404
  digitalWrite(S1_OUT, s1);
  digitalWrite(S2_OUT, s2);
  digitalWrite(S3_OUT, s3);
  digitalWrite(S4_OUT, s4);

  // Detect car at gate
  long distance = getDistance();

  bool carDetected =
    (distance > 0 && distance <= GATE_THRESHOLD);

  // Send CAR signal to 7408
  digitalWrite(CAR_OUT,
               carDetected ? HIGH : LOW);

  // Read FULL signal from 7408
  bool parkingFull =
    (digitalRead(FULL_IN) == HIGH);

  // ==========================================================
  // FULL PARKING
  // ==========================================================

  if (parkingFull) {

    // Keep barrier closed
    barrier.write(SERVO_CLOSED);

    // Buzzer only if car detected
    if (carDetected) {

      digitalWrite(BUZZER_PIN, HIGH);

    } else {

      digitalWrite(BUZZER_PIN, LOW);
    }
  }

  // ==========================================================
  // PARKING AVAILABLE
  // ==========================================================

  else {

    digitalWrite(BUZZER_PIN, LOW);

    if (carDetected) {

      // Open barrier
      barrier.write(SERVO_OPEN);

      delay(BARRIER_DELAY);

      // Close barrier
      barrier.write(SERVO_CLOSED);
    }

    else {

      barrier.write(SERVO_CLOSED);
    }
  }

  delay(200);
}

// ============================================================
// ULTRASONIC DISTANCE FUNCTION
// ============================================================

long getDistance() {

  digitalWrite(TRIG, LOW);
  delayMicroseconds(2);

  digitalWrite(TRIG, HIGH);
  delayMicroseconds(10);

  digitalWrite(TRIG, LOW);

  long duration =
    pulseIn(ECHO, HIGH, 30000);

  if (duration == 0)
    return -1;

  return duration / 58;
}
